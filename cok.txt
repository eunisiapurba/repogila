class Hotel:
    def __init__(self):
        self.kamar_tersedia = {101: 300000, 102: 500000, 103: 1000000}

    def tampilkan_kamar_tersedia(self):
        print("\nKamar yang tersedia:")
        for kamar, harga in self.kamar_tersedia.items():
            print(f"Kamar {kamar} - Rp {harga}/malam")

    def pesan_kamar(self, nomor_kamar):
        if nomor_kamar in self.kamar_tersedia:
            del self.kamar_tersedia[nomor_kamar]
            print(f"Kamar {nomor_kamar} berhasil dipesan.")
        else:
            print(f"Kamar {nomor_kamar} tidak tersedia.")

    def checkout_kamar(self, nomor_kamar, harga_kamar):
        self.kamar_tersedia[nomor_kamar] = harga_kamar
        print(f"Kamar {nomor_kamar} berhasil di-checkout dan tersedia kembali.")

# Contoh penggunaan
hotelku = Hotel()

# Tampilkan kamar yang tersedia
hotelku.tampilkan_kamar_tersedia()

# Pesan kamar
hotelku.pesan_kamar(102)

# Tampilkan kamar setelah pemesanan
hotelku.tampilkan_kamar_tersedia()

# Checkout kamar
hotelku.checkout_kamar(102, 500000)

# Tampilkan kamar setelah checkout
hotelku.tampilkan_kamar_tersedia()
